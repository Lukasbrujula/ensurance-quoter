### 4. Implementation

#### File 1: `lib/google/oauth.ts`

```typescript
// OAuth client factory
// - Creates OAuth2Client from googleapis
// - Accepts stored tokens
// - Handles auto-refresh via 'tokens' event listener
// - Exports: createOAuthClient(tokens), generateAuthUrl(userId), exchangeCode(code)

import { google } from 'googleapis';

export function getOAuth2Client() {
  return new google.auth.OAuth2(
    process.env.GOOGLE_CLIENT_ID,
    process.env.GOOGLE_CLIENT_SECRET,
    process.env.GOOGLE_REDIRECT_URI,
  );
}

export function generateAuthUrl(userId: string): string {
  const client = getOAuth2Client();
  return client.generateAuthUrl({
    access_type: 'offline',
    prompt: 'consent',
    scope: ['https://www.googleapis.com/auth/calendar.events'],
    state: userId,
  });
}

export async function exchangeCodeForTokens(code: string) {
  const client = getOAuth2Client();
  const { tokens } = await client.getToken(code);
  return tokens;
}

export function createAuthenticatedClient(tokens: {
  access_token: string;
  refresh_token: string;
  expiry_date: number;
}) {
  const client = getOAuth2Client();
  client.setCredentials(tokens);
  return client;
}
```

#### File 2: `lib/google/calendar-service.ts`

Five exported functions:
```typescript
export async function createCalendarEvent(agentId, event) → string | null  // returns Google event ID
export async function updateCalendarEvent(agentId, googleEventId, updates) → boolean
export async function deleteCalendarEvent(agentId, googleEventId) → boolean
export async function getUpcomingEvents(agentId, days = 7) → CalendarEvent[]
export async function isCalendarConnected(agentId) → boolean
```

Each function:
1. Gets stored tokens from DB via `getGoogleTokens(agentId)`
2. If no tokens, returns null/false/empty (gracefully)
3. Creates authenticated client via `createAuthenticatedClient(tokens)`
4. Calls Google Calendar API
5. Handles token refresh (listen for 'tokens' event, update DB)
6. Wraps in try/catch — calendar errors never crash the caller

Event creation includes:
- `extendedProperties.private.source = 'ensurance'` — marks our events
- `extendedProperties.private.ensurance_lead_id` — links back to lead
- Reminders: 15 min + 5 min popup notifications

#### Database migration:

```sql
-- New table for Google OAuth tokens
CREATE TABLE IF NOT EXISTS google_integrations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  access_token TEXT NOT NULL,
  refresh_token TEXT NOT NULL,
  token_expiry TIMESTAMPTZ NOT NULL,
  email TEXT,
  calendar_id TEXT DEFAULT 'primary',
  connected_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(agent_id)
);

-- RLS
ALTER TABLE google_integrations ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own google integration"
  ON google_integrations FOR ALL
  USING (agent_id = auth.uid())
  WITH CHECK (agent_id = auth.uid());

-- Index
CREATE INDEX idx_google_integrations_agent ON google_integrations(agent_id);

-- Add google_event_id to leads for linking
ALTER TABLE leads ADD COLUMN IF NOT EXISTS google_event_id TEXT;
```

### 6. Failure Handling
- If `GOOGLE_CLIENT_ID` is not set, the "Connect" button should be disabled with tooltip: "Google Calendar not configured"
- If token exchange fails, redirect to integrations with `?google=error` and show toast
- If token refresh fails during a calendar operation, mark as disconnected in DB and return gracefully
- If Google revokes access externally, next API call will fail — catch and mark as disconnected

### 7. Guardrails
- Do NOT store tokens in plain text in client-accessible storage (database with RLS only)
- Do NOT expose tokens to the frontend (only the connected email and status)
- Do NOT call Google Calendar API from client components (server-side only)
- Do NOT install any package other than `googleapis`
- All Google API calls are server-side, behind auth guard
- Rate limit the OAuth routes (use the `auth` tier rate limiter)
